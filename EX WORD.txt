CREATE TABLE CLIENTE
(
	COD INTEGER PRIMARY KEY,
	NOME VARCHAR (50),
	RUA VARCHAR (30),
	NRO INTEGER,
	CEP INTEGER,
	CIDADE VARCHAR (40)
);

CREATE TABLE TELEFONECLIENTE
(
	TELEFONE VARCHAR (13),
	COD INTEGER,
	CONSTRAINT FK_COD FOREIGN KEY (COD) REFERENCES CLIENTE
);

CREATE TABLE EXEMPLAR
(
	NROEXEMPLAR INTEGER PRIMARY KEY,
	CODFILME INTEGER,
	CONSTRAINT FK_CODFILME FOREIGN KEY (CODFILME) REFERENCES FILME
);

CREATE TABLE FILME
(
	CODFILME INTEGER,
	NOME VARCHAR (50),
	GENERO VARCHAR (20),
	DTALANC DATE,
	CONSTRAINT PK_CODFILME PRIMARY KEY (CODFILME)
);

CREATE TABLE ATOR
(
	IDATOR INTEGER,
	NOME VARCHAR (50),
	DTANASC DATE,
	CONSTRAINT PK_ATOR PRIMARY KEY (IDATOR)
);

CREATE TABLE ATORESTRELAFILME
(
	IDATOR INTEGER,
	CODFILME INTEGER,
	CONSTRAINT FK_IDATOR FOREIGN KEY (IDATOR) REFERENCES ATOR,
	CONSTRAINT FK_CODFILME FOREIGN KEY (CODFILME) REFERENCES FILME
);

CREATE TABLE CLIENTEALUGAEXEMPLARFILME
(
	CODCLIENTE INTEGER,
	CODFILME INTEGER,
	NROEXEMPLAR INTEGER,
	DTARET DATE,
	DTADEV DATE,
	CONSTRAINT FK_CODCLIENTE FOREIGN KEY (CODCLIENTE) REFERENCES CLIENTE,
	CONSTRAINT FK_CODFILME FOREIGN KEY (CODFILME) REFERENCES FILME,
	CONSTRAINT FK_NROEXEMPLAR FOREIGN KEY (NROEXEMPLAR) REFERENCES EXEMPLAR
);
